const express = require("express");
const { Dropbox } = require("dropbox");
const fs = require("fs");

const app = express();
const dbx = new Dropbox({ accessToken: "YOUR_DROPBOX_ACCESS_TOKEN" });

const FILE_NAME = "example.txt"; // file in same folder

fs.readFile(FILE_NAME, (err, contents) => {
  if (err) throw err;
  dbx.filesUpload({ path: "/" + FILE_NAME, contents })
    .then(() => console.log("File uploaded!"))
    .catch(err => console.error(err));
});

app.listen(3000, () => console.log("Running..."));


download,js

const express = require("express");
const { Dropbox } = require("dropbox");
const fs = require("fs");

const app = express();
const dbx = new Dropbox({ accessToken: "YOUR_DROPBOX_ACCESS_TOKEN" });

const FILE_NAME = "example.txt"; // file to download from Dropbox

dbx.filesDownload({ path: "/" + FILE_NAME })
  .then(res => {
    fs.writeFileSync(FILE_NAME, res.result.fileBinary, "binary");
    console.log("File downloaded!");
  })
  .catch(err => console.error(err));

app.listen(3000, () => console.log("Running..."));
