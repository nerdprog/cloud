<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Upload Image to Dropbox</title>
</head>
<body style="font-family: Arial; padding: 30px;">
  <h2>Upload Image to Dropbox</h2>

  <input type="file" id="fileInput" accept="image/*">
  <button id="uploadBtn">Upload</button>

  <p id="status"></p>

  <script>
    // ⚠️ Replace with your real Dropbox access token (for testing only)
    const ACCESS_TOKEN = "YOUR_DROPBOX_ACCESS_TOKEN";

    document.getElementById("uploadBtn").addEventListener("click", async () => {
      const fileInput = document.getElementById("fileInput");
      const status = document.getElementById("status");

      if (!fileInput.files.length) {
        status.textContent = "Please select a file first.";
        return;
      }

      const file = fileInput.files[0];
      status.textContent = "Uploading...";

      try {
        const res = await fetch("https://content.dropboxapi.com/2/files/upload", {
          method: "POST",
          headers: {
            "Authorization": "Bearer " + ACCESS_TOKEN,
            "Dropbox-API-Arg": JSON.stringify({
              path: "/" + file.name,
              mode: "add",
              autorename: true,
              mute: false
            }),
            "Content-Type": "application/octet-stream"
          },
          body: file
        });

        if (res.ok) {
          const data = await res.json();
          status.textContent = "✅ Upload successful: " + data.name;
        } else {
          const err = await res.text();
          status.textContent = "❌ Upload failed: " + err;
        }
      } catch (error) {
        status.textContent = "Error: " + error.message;
      }
    });
  </script>
</body>
</html>
